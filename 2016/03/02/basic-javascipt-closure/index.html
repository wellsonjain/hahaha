Living in dreams of yesterday, we find ourselves still dreaming of impossible future conquests - Charles Lindbergh We mentioned about function:

function buildTicket (allRides, passRides, pick){
  if (passRides[0] == pick) {
    var pass = allRides.shift()
    return function () { alert("Quick! You've got a Fast Pass to" + pass + "!") }
  } else {
    for(var i = 0; i<allRides.length; i++){
      if(allRides[i][0] == pick){
        return function () { alert("A ticket is printing for " + pick + "!\n" + "Your wait time is about " + allRides[i][1] + " minutes."); };
      }
    }
  }
}

As you can see, returning a function from a function, complete with variables from an external scope, is called a closure. The entire contents of one of these inner functions will still be available OUTSIDE the outermost function.

Here is an example,

function testClosure ( ) {
  var x = 4; // Local variable
  return x;
}

testClosure(); // -> 4
x; // -> undefined

x is local variable, they aren’t available once the function’s scope is closed! And what if we make this function more complicate.

function testClosure ( ) {
  var x = 4;
  function closeX ( ) {
    return x;
  }
  return closeX;
}

x is not declare in closeX(), but when we call the testClosure(), we still get the result 4. Why was that? That’s because the inner function can access the outer function’s variables, because they “feel” like global variables.

We can also modify bound variables in the background

function buildCoveTicketMaker( transport ) {
  var passengerNumber = 0;

    passengerNumber++;

In the end, it’s gonna talk about some common mistakes in loops with closure. Javascript has different way to load program, which is called “Hoist”. Memory is set aside of all necessary variables and declared functions first.

Here is an example,

function sumOfSquares (a, b){
  var x = add(a*a, b*b);
  return x;  

  function add (c, d){
    var a = c + d;
    return a;
  }
}

Infact, Javascript loads like this,

function sumOfSquares (a, b){
  var x = undefined;
  function add (c, d){
    var a = c + d;
    return a;
  }

  x = add(a*a, b*b);
  return x;
}

The necessary variable and function will be hoisted before the operational code run.

More example:

function getMysteryNumber () {

  function chooseMystery() {
    return 12;
  }

  return chooseMystery();

  function chooseMystery() {
    return 7;
  }
}
// -> 7
// The second chooseMystery() will override the first one.

function getMysteryNumber () {
  var chooseMystery = function() {
    return 12;
  }

  return chooseMystery();

  var chooseMystery = function() {
    return 7;
  }
}
// -> 12

This is a little bit confusing. That is because function expressions are never hoisted. It will look like this in Javascript.

function getMysteryNumber () {
  var chooseMystery = undefined;
  var chooseMystery = undefined; // Override the first one

  chooseMystery = function() {
    return 12;
  }

  return chooseMystery(); // -> 12

  // unreachable
  chooseMystery = function() {
    return 7;
  }
}

One more example:

function capacityStatus (numPassengers, capacity) {

  if (numPassengers == capacity) {
    noSeats();
  } else {
    seatsAvail();

If we call the function by capacityStatus(60, 60), the result will be ERROR. But it seems looking good. Let’s take a look what the inside of javascript loading.

function capacityStatus (numPassengers, capacity) {
  var noSeats = undefined;
  var seatsAvail = undefined;

  if (numPassengers == capacity) {
    noSeats(); // Not exist yet
  } else {
    seatsAvail();
  }

  noSeats = function (){
    alert("No seats left!");   
    return false;
  }

  seatsAvail = function(){
    alert("There are " + (capacity - numPassengers) + " seats left!");   
    return true;
  }
}

noSeat() is called when it’s not define yet. There are two way to fix the problem.

// First One: Change the order, and the function will be called after being defined.
function capacityStatus (numPassengers, capacity) {

  var noSeats = function (){
    alert("No seats left!");   
    return false;
  }

  var seatsAvail = function(){
    alert("There are " + (capacity - numPassengers) + " seats left!");   
    return true;
  }

  if (numPassengers == capacity) {
    noSeats();
  } else {
    seatsAvail();
  }
}

// Second: Do not ues function expression, use declartion.
function capacityStatus (numPassengers, capacity) {

  if (numPassengers == capacity) {
    noSeats(); // Not exist yet
  } else {
    seatsAvail();
  }

  function noSeats(){
    alert("No seats left!");   
    return false;
  }

  function seatsAvail(){
    alert("There are " + (capacity - numPassengers) + " seats left!");   
    return true;
  }
}
